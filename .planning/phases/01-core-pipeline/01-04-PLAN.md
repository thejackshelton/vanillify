```typescript
import type { NodeEntry, Warning } from '../types'
import type { NameMap } from './namer'
import { selectorFor } from './namer'
import { generateCSS } from './generator'

export interface RewriteResult {
  /** Source with className values replaced by indexed names */
  component: string
  /** Generated vanilla CSS with .nodeN selectors */
  css: string
  /** All warnings (from extraction + generation) */
  warnings: Warning[]
}

/**
 * Rewrite the source component and assemble per-node CSS.
 *
 * For each static entry:
 * 1. Generate CSS for that node's tokens individually
 * 2. Replace UnoCSS-generated selectors with .nodeN
 * 3. Replace the className value in the source with "nodeN"
 *
 * Dynamic entries are left unchanged in the source.
 * Replacements applied in reverse source order to avoid offset drift.
 *
 * @param source - Original JSX/TSX source string
 * @param entries - NodeEntry[] from extractor
 * @param nameMap - NameMap from namer
 * @param extractWarnings - Warnings from extraction phase
 * @returns RewriteResult with transformed component, CSS, and all warnings
 */
export async function rewrite(
  source: string,
  entries: NodeEntry[],
  nameMap: NameMap,
  extractWarnings: Warning[]
): Promise<RewriteResult> {
  const allWarnings: Warning[] = [...extractWarnings]
  const cssBlocks: string[] = []

  // Generate CSS per-node to get isolated CSS blocks
  for (const entry of entries) {
    if (entry.isDynamic) continue
    const name = nameMap.get(entry.nodeIndex)
    if (!name) continue

    const tokens = new Set(entry.classNames)
    const result = await generateCSS(tokens)

    // Collect unmatched warnings
    allWarnings.push(...result.warnings)

    // Replace UnoCSS selectors with .nodeN
    // UnoCSS generates selectors like .flex, .bg-red-500, etc.
    // We need to replace them all with .nodeN for this node
    const nodeCSS = buildNodeCSS(name, result.css, entry.classNames)
    if (nodeCSS.trim()) {
      cssBlocks.push(nodeCSS)
    }
  }

  // Rewrite source -- replace className values with indexed names
  // Sort replacements in reverse order to avoid offset drift
  const replacements = entries
    .filter(e => !e.isDynamic && nameMap.has(e.nodeIndex))
    .map(e => ({
      span: e.span,
      newValue: `"${nameMap.get(e.nodeIndex)}"`,
    }))
    .sort((a, b) => b.span.start - a.span.start)

  let component = source
  for (const { span, newValue } of replacements) {
    component = component.slice(0, span.start) + newValue + component.slice(span.end)
  }

  const css = cssBlocks.join('\n\n')

  return { component, css, warnings: allWarnings }
}
```

2. Implement the `buildNodeCSS` helper:

```typescript
/**
 * Transform UnoCSS-generated CSS to use .nodeN selectors.
 *
 * Strategy: Parse the CSS text to find rule blocks and replace
 * the utility class selectors with the node selector.
 *
 * For plain rules: .utility { props } -> .nodeN { props }
 * For @media rules: @media (...) { .utility { props } } -> @media (...) { .nodeN { props } }
 * For pseudo rules: .utility:hover { props } -> .nodeN:hover { props }
 *
 * Since UnoCSS generates one rule per utility, we need to:
 * 1. Extract all CSS declarations from the generated rules
 * 2. Group plain declarations under .nodeN { ... }
 * 3. Keep @media and pseudo-selector rules separate with .nodeN
 */
function buildNodeCSS(nodeName: string, rawCSS: string, classNames: string[]): string {
  const selector = `.${nodeName}`
  const lines = rawCSS.split('\n')
  const plainDeclarations: string[] = []
  const specialBlocks: string[] = []

  let i = 0
  while (i < lines.length) {
    const line = lines[i].trim()

    // Skip empty lines
    if (!line) { i++; continue }

    // @media or @supports block
    if (line.startsWith('@media') || line.startsWith('@supports')) {
      const block = collectBlock(lines, i)
      // Replace utility selectors with node selector inside the block
      const rewritten = rewriteSelectorsInBlock(block, classNames, selector)
      specialBlocks.push(rewritten)
      i += countLines(block)
      continue
    }

    // Regular rule: .utility { declaration }
    if (line.startsWith('.') && line.includes('{')) {
      const block = collectBlock(lines, i)
      const declarations = extractDeclarations(block)
      const pseudoMatch = line.match(/\.[^\s{]+?(:[^\s{]+)\s*\{/)
      if (pseudoMatch) {
        // Pseudo-selector rule: .utility:hover { ... } -> .nodeN:hover { ... }
        specialBlocks.push(`${selector}${pseudoMatch[1]} {\n${declarations}\n}`)
      } else {
        // Plain rule -- collect declarations for merging
        plainDeclarations.push(declarations)
      }
      i += countLines(block)
      continue
    }

    i++
  }

  const parts: string[] = []
  if (plainDeclarations.length > 0) {
    parts.push(`${selector} {\n${plainDeclarations.join('\n')}\n}`)
  }
  parts.push(...specialBlocks)

  return parts.join('\n\n')
}
```

Note: The helper functions `collectBlock`, `countLines`, `extractDeclarations`, and `rewriteSelectorsInBlock` are CSS text manipulation utilities. Implement them as simple line-based parsers that track brace nesting depth. These do NOT need to be full CSS parsers -- UnoCSS output is predictable and well-formatted.

Key implementation notes:
- `collectBlock(lines, startIndex)`: Collect lines from start until braces balance (depth returns to 0). Return as single string.
- `countLines(block)`: Count newlines + 1 in the block string.
- `extractDeclarations(block)`: Return the lines between the first `{` and last `}`, trimmed.
- `rewriteSelectorsInBlock(block, classNames, selector)`: Within a @media block, find inner selectors that match any of the classNames utilities and replace with the node selector. Handle escaped characters in CSS selectors.

IMPORTANT: UnoCSS escapes special characters in selectors. For example, `text-[#ff0000]` becomes `.text-\[\#ff0000\]` in CSS. The selector replacement must handle these escapes. Use a regex that matches any of the generated utility selectors and replaces with `.nodeN`.

```typescript
import { parse } from './pipeline/parser'
import { extract } from './pipeline/extractor'
import { generateCSS } from './pipeline/generator'
import { assignNames } from './pipeline/namer'
import { rewrite } from './pipeline/rewriter'
import type { ConvertOptions, ConvertResult } from './types'

/**
 * Convert a JSX/TSX source file from Tailwind classes to vanilla CSS.
 *
 * Accepts source code as a string, extracts Tailwind class names from
 * className/class attributes using AST parsing (not regex), generates
 * vanilla CSS using UnoCSS's createGenerator with preset-wind4, and
 * returns the transformed component with indexed class names (.node0, .node1)
 * plus the generated CSS.
 *
 * @param source - JSX/TSX source code string
 * @param filename - Filename for parser language detection (e.g., "component.tsx")
 * @param options - Optional configuration (reserved for Phase 2 customVariants)
 * @returns Promise<ConvertResult> with { component, css, warnings }
 */
export async function convert(
  source: string,
  filename: string,
  options?: ConvertOptions
): Promise<ConvertResult> {
  // 1. Parse source to AST
  const { program } = parse(filename, source)

  // 2. Extract class entries from AST
  const { entries, warnings: extractWarnings } = extract(program, source)

  // 3. Assign indexed class names
  const nameMap = assignNames(entries)

  // 4. Rewrite source and generate per-node CSS
  const result = await rewrite(source, entries, nameMap, extractWarnings)

  return result
}

export type { ConvertOptions, ConvertResult, Warning, NodeEntry } from './types'
```
