1. Create src/pipeline/parser.ts:
   ```typescript
   import { parseSync } from 'oxc-parser'

   export interface ParseResult {
     program: any  // ESTree Program node from oxc-parser
     errors: Array<{ message: string }>
   }

   /**
    * Parse JSX/TSX source into an ESTree-compatible AST.
    * Language is inferred from the filename extension by oxc-parser.
    *
    * @param filename - Used for language detection (e.g., "component.tsx")
    * @param source - The source code string to parse
    * @returns ParseResult with the AST program node
    * @throws Error if parsing produces errors
    */
   export function parse(filename: string, source: string): ParseResult {
     const result = parseSync(filename, source, {
       sourceType: 'module',
       lang: filename.endsWith('.tsx') || filename.endsWith('.ts') ? 'tsx' : 'jsx',
     })

     if (result.errors && result.errors.length > 0) {
       throw new Error(
         `Parse error in ${filename}: ${result.errors.map((e: any) => e.message).join(', ')}`
       )
     }

     return {
       program: result.program,
       errors: [],
     }
   }
   ```

   Note: The `lang` option may need adjustment based on oxc-parser's actual API. The parseSync signature is `parseSync(filename, sourceText, options?)`. Check if `lang` is needed or if filename extension is sufficient. If parseSync infers language from filename, remove the explicit `lang` option.

2. Create src/pipeline/parser.test.ts:

     it('parses a simple TSX file into an AST with JSXElement', () => {
       const source = '<div className="flex">hello</div>'
       const result = parse('test.tsx', source)
       expect(result.program).toBeDefined()

     })

           <div className="flex items-center">
             <span className="text-red-500">hi</span>
           </div>
         )
       `
       const result = parse('test.tsx', source)
       expect(result.program.type).toBe('Program')
     })

     it('throws on invalid syntax', () => {
       const source = '<div className='

     })

     it('handles class attribute (not just className)', () => {
       const source = '<div class="flex gap-2">content</div>'
